//Биномиальное число по полотну Кулешова
// Для Вычисления настоящего Бинома =>
// RightBin(a,b) = KulBin(a-b+1,b)
//Например: RightBin(10,3) = KulBin(10-3+1,3) = KulBin(8,3)
double KulBin(double a, double b){
if(a<=0){return 0;}
double ans=1,div=1;
for(int i=1;i<b+1;i++){div*=i;}
for(int i=a;i<a+b;i++){ans*=i;}
return ans/div;
}
